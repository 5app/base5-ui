# InlineList

Lay out children horizontally with optional spacing or separators.

Supports styleProps `position`, `text`, and `size`.

<Playground>
	<InlineList spacing="s">
		<Icon name="asset-image" />
		<Icon name="asset-podcast" />
		<Icon name="asset-audio" />
	</InlineList>

    <InlineList splitBy="comma">
    	<>Milk</>
    	<>Flour</>
    	<>Sugar</>
    	<>Eggs</>
    </InlineList>

    <InlineList splitBy="dot">
    	<TextLink href="#">Home</TextLink>
    	<TextLink href="#">Projects</TextLink>
    	<TextLink href="#">About us</TextLink>
    </InlineList>

</Playground>

<Props of={InlineList} />
