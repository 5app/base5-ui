# Combobox

An accessible combobox component that can be used with any input element.

Implemented based on the recommendations from the [WAI-ARIA Authoring Practices 1.2 working draft](https://www.w3.org/TR/wai-aria-practices-1.2/#combobox).

## Examples

<Playground>
	{() => {
		const defaultOptions = [
			'Apple',
			'Apricot',
			'Banana',
			'Blueberry',
			'Cherry',
			'Fig',
			'Feijoa',
			'Gooseberry',
			'Jackfruit',
			'Jujube',
			'Kiwi',
			'Kumquat',
			'Lingonberry',
			'Lychee',
			'Mandarin',
			'Mango',
			'Melon',
			'Nectarine',
			'Orange',
			'Papaya',
			'Passion Fruit',
			'Pear',
			'Persimmon',
			'Physalis',
			'Pineapple',
			'Plum',
			'Pomegranate',
			'Pomelo',
			'Raspberry',
			'Starfruit',
			'Strawberry',
			'Watermelon',
			'Wumpa',
		];
		const inputRef = React.useRef();
		const [value, setValue] = React.useState('');
		const [options, setOptions] = React.useState(defaultOptions);
		React.useEffect(() => {
			const filteredOptions = defaultOptions.filter(option =>
				option.toLowerCase().includes(value.toLowerCase())
			);
			setOptions(filteredOptions);
		}, [value]);
		return (
			<form>
				<label htmlFor="fruit-selector" style={{display: 'block'}}>
					Pick a fruit
				</label>
				<Combobox
					id="fruit-selector"
					inputValue={value}
					resultCount={options.length}
					shouldShowStatusMessage={!options.length}
					onSelect={selectedItem => {
						setValue(selectedItem.value);
						inputRef.current.focus();
					}}
					onInputChange={e => setValue(e.target.value)}
				>
					<ComboboxInput ref={inputRef} />
					<ComboboxMenu aria-label="Fruits">
						{options.map(item => (
							<ComboboxMenuItem key={item}>
								{item}
							</ComboboxMenuItem>
						))}
					</ComboboxMenu>
				</Combobox>
			</form>
		);
	}}
</Playground>
